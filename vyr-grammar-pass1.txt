program:
  item_list

item_list:
  item item_list_opt

item:
  fn_decl
  struct_decl
  enum_decl
  trait_decl
  impl_decl
  mod_decl
  use_decl
  const_decl

mod_decl:
  mod IDENT { item_list_opt }

use_decl:
  use path ;

const_decl:
  const IDENT : type = expr ;

fn_decl:
  fn IDENT generic_params_opt param_list return_type_opt where_clause_opt block

generic_params:
  < generic_param generic_param_tail_opt >

generic_param_tail:
  , generic_param generic_param_tail_opt

generic_param:
  IDENT

param_list:
  ( params_opt )

params:
  param param_tail_opt

param_tail:
  , param param_tail_opt

param:
  IDENT : type

return_type:
  -> type

where_clause:
  where where_pred where_pred_tail_opt

where_pred_tail:
  , where_pred where_pred_tail_opt

where_pred:
  type : trait_bound_list

trait_bound_list:
  path trait_bound_tail_opt

trait_bound_tail:
  + path trait_bound_tail_opt

struct_decl:
  struct IDENT generic_params_opt struct_body

struct_body:
  { field_list_opt }

field_list:
  field field_tail_opt

field_tail:
  , field field_tail_opt

field:
  IDENT : type

enum_decl:
  enum IDENT generic_params_opt { variant_list_opt }

variant_list:
  variant variant_tail_opt

variant_tail:
  , variant variant_tail_opt

variant:
  IDENT variant_payload_opt

variant_payload:
  ( type_list_opt )

type_list:
  type type_tail_opt

type_tail:
  , type type_tail_opt

trait_decl:
  trait IDENT generic_params_opt where_clause_opt { trait_item_list_opt }

trait_item_list:
  trait_item trait_item_list_opt

trait_item:
  fn IDENT generic_params_opt param_list return_type_opt where_clause_opt ;

impl_decl:
  impl generic_params_opt impl_target where_clause_opt { impl_item_list_opt }

impl_target:
  type for type
  type

impl_item_list:
  impl_item impl_item_list_opt

impl_item:
  fn_decl
  const_decl

type:
  type_atom type_suffix_opt

type_suffix:
  < type_list >

type_atom:
  path
  & type
  & mut type
  * type
  ( type )

path:
  IDENT path_tail_opt

path_tail:
  :: IDENT path_tail_opt

block:
  { stmt_list_opt expr_opt }

stmt_list:
  stmt stmt_list_opt

stmt:
  let_stmt
  semi_stmt

let_stmt:
  let mut_opt IDENT type_ascription_opt = expr ;

type_ascription:
  : type

semi_stmt:
  expr ;

expr:
  if_expr
  loop_expr
  match_expr
  assign_expr

assign_expr:
  logic_or_expr
  logic_or_expr = expr

logic_or_expr:
  logic_and_expr logic_or_tail_opt

logic_or_tail:
  || logic_and_expr logic_or_tail_opt

logic_and_expr:
  equality_expr logic_and_tail_opt

logic_and_tail:
  && equality_expr logic_and_tail_opt

equality_expr:
  additive_expr equality_tail_opt

equality_tail:
  == additive_expr equality_tail_opt
  != additive_expr equality_tail_opt

additive_expr:
  multiplicative_expr additive_tail_opt

additive_tail:
  + multiplicative_expr additive_tail_opt
  - multiplicative_expr additive_tail_opt

multiplicative_expr:
  unary_expr multiplicative_tail_opt

multiplicative_tail:
  * unary_expr multiplicative_tail_opt
  / unary_expr multiplicative_tail_opt
  % unary_expr multiplicative_tail_opt

unary_expr:
  ! unary_expr
  - unary_expr
  call_expr

call_expr:
  primary_expr call_suffix_opt

call_suffix:
  ( arg_list_opt ) call_suffix_opt
  . IDENT call_suffix_opt

arg_list:
  expr arg_tail_opt

arg_tail:
  , expr arg_tail_opt

primary_expr:
  literal
  path
  ( expr )
  block

if_expr:
  if expr block else_branch_opt

else_branch:
  else block
  else if_expr

loop_expr:
  loop block
  while expr block

match_expr:
  match expr { match_arm_list_opt }

match_arm_list:
  match_arm match_arm_list_opt

match_arm:
  pattern -> expr comma_opt

comma:
  ,

pattern:
  IDENT
  _
  literal

literal:
  INT_LIT
  FLOAT_LIT
  STRING_LIT
  BOOL_LIT
